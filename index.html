FRB.js

Update the 'update:' function, and press cmd+b (or ctrl+b if on windows).

var emitter = { pool: new frb.ResourcePool(), position: {x:0, y:0}, items: function() { return this.pool.active; }, emit: function() { var self = this; var newParticle = this.pool.add(function() { return frb.SpriteManager.add(self.texture) }); newParticle.alpha = 1; newParticle.x = this.position.x; newParticle.y = this.position.y; return newParticle; } }; window.emitter = emitter; frb.start({ init: function() { emitter.texture = "Frblogo128"; var circle = frb.SpriteManager.addCircle(10); circle.xVelocity = 100; }, update: function() { emitter.position.x = frb.InputManager.mouse.worldX; emitter.position.y = frb.InputManager.mouse.worldY; // recycle particles if necessary var list = emitter.items(); for (var i = 0; i < list.length; i++) { var particle = list.get(i); if (particle.alpha <= 0) { emitter.pool.return(particle); continue; } particle.alpha -= 0.05; }; var newParticle = emitter.emit(); // initialize particle var xRange = 100; var gravity = -150.5; var direction = {x: (Math.random() * xRange) - (xRange/2), y: Math.random() * 400}; newParticle.xVelocity = direction.x; newParticle.yVelocity = direction.y; newParticle.yAcceleration = gravity; } });
